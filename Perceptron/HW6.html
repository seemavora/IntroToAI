Y = pd.DataFrame(Y)
Y_restruct = pd.get_dummies(Y, columns = Y.columns, prefix = Y.columns)

for i in range(0,len(numbers["target_names"])):
    np.set_printoptions(precision=2)

    X_train, X_test, Y_train, Y_test = train_test_split(X, Y_restruct['0_' + str(i)], test_size=0.2, random_state=0, stratify=Y)
    model = Perceptron(tol=1e-3, random_state=0)
    model.fit(X_train, Y_train)

    disp = plot_confusion_matrix(model, X_test, Y_test,
                                display_labels=['Not ' + str(i), str(i)],
                                cmap=plt.cm.Blues)
    accuracy = model.score(X_test, Y_test)
    disp.ax_.set_title("Test results for %s (%s) " % (i, accuracy))
    plt.show()
    print('Accuracy of perceptron on training set: {:.2f}'.format(model.score(X_train, Y_train)))

    print('Accuracy of perceptron on test set: {:.2f}'.format(model.score(X_test, Y_test)))
